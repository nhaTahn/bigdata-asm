# 0) Clean toàn bộ
make down
make reset-stream
: > data/replay/events.jsonl
rm -f logs/bronze-silver.pid logs/replay-full.pid logs/replay-full.log

# 1) Kafka + topic sạch
make create-topics
podman compose exec -T kafka kafka-topics --bootstrap-server kafka:29092 --delete --if-exists --topic ${KAFKA_TOPIC_RAW_GPS:-bus-gps-raw}
sleep 2
podman compose exec -T kafka kafka-topics --bootstrap-server kafka:29092 --create --if-not-exists --topic ${KAFKA_TOPIC_RAW_GPS:-bus-gps-raw} --partitions 1 --replication-factor 1

# 2) Start Bronze/Silver streaming (nền)
make bronze-silver-bg SPARK_LOCAL_CORES=1 SPARK_DRIVER_MEMORY=1g SPARK_EXECUTOR_MEMORY=1g SPARK_SHUFFLE_PARTITIONS=1 SPARK_DEFAULT_PARALLELISM=1

# 3) Start replay FULL (nền)
nohup bash -lc '. .venv/bin/activate && REPLAY_MODE=full python -m producer.replay_producer --config configs/pipeline.yaml' > logs/replay-full.log 2>&1 & echo $! > logs/replay-full.pid


ps -p $(cat logs/bronze-silver.pid) -o pid,stat,etime,command
ps -p $(cat logs/replay-full.pid) -o pid,stat,etime,command
tail -f logs/replay-full.log
tail -f logs/bronze-silver.log
wc -l data/replay/events.jsonl


Khi replay-full.log có dòng Replay completed, chạy tiếp:

make gold SPARK_LOCAL_CORES=1 SPARK_DRIVER_MEMORY=1g SPARK_EXECUTOR_MEMORY=1g SPARK_SHUFFLE_PARTITIONS=1 SPARK_DEFAULT_PARALLELISM=1
make report-assets
open docs/dashboard/index.html
